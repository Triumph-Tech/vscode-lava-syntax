<Row>
  <Column>

|    Prefix | Method                                                                                  |
| --------: | --------------------------------------------------------------------------------------- |
|      `if` | <code>{% if condition %} ... {% endif %}</code>                                                    |

|  `elseif` | <code>{% elseif condition %}</code>                                                                |
|  `ifelse` | <code>{% if condition %} ... {% else %} ... {% endif %}</code>                                     |
|  `unless` | <code>{% unless condition %} ... {% endunless %}</code>                                            |
|    `case` | <code>{% case variable %} ... {% when condition %} ... {% else %} ... {% endcase %}</code>         |

| `ifelse-` | <code>{%- if condition -%} ... {%- else -%} ... {%- endif -%}</code>                               |
| `unless-` | <code>{%- unless condition -%} ... {%- endunless -%}</code>                                        |
|   `case-` | <code>{%- case variable -%} ... {%- when condition -%} ... {%- else -%} ... {%- endcase -%}</code> |
